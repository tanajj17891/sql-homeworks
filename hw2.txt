select p.firstName, p.lastName, a.city, a.state

question 6)
SELECT c.name AS customers
FROM Customers c
LEFT JOIN Orders o
    ON c.id = o.customerId
WHERE o.id IS NULL;

question 7)
select e.name, b.bonus
from Employee e
left join Bonus b on e.empId = b.empId 
where b.bonus < 1000 or b.bonus is null
